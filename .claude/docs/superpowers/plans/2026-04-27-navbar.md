# Navbar Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add a fixed top navigation bar with brand left, scroll-spy nav links center, and action icon buttons right; transparent over hero, frosted on scroll; mobile hamburger with fullscreen overlay.

**Architecture:** Single `Navbar.astro` component with inline `<script>` for all interactivity (scroll behavior, scroll-spy via IntersectionObserver, mobile overlay, theme toggle). `BrandLogo.astro` extracted as a reusable sub-component. `ThemeToggle.astro` removed — its logic moves into `Navbar.astro`. Three new SVG icons added to `Icon.astro`.

**Tech Stack:** Astro, Tailwind CSS v4, vanilla JS (IntersectionObserver, scroll events), CSS custom properties from existing design system.

<script>
  // ── Scroll: transparent → frosted ───────────────────
  const navbar = document.getElementById('navbar');

  function updateScrolled() {
    if (window.scrollY > 20) {
      navbar?.classList.add('scrolled');
    } else {
      navbar?.classList.remove('scrolled');
    }
  }

  window.addEventListener('scroll', updateScrolled, { passive: true });
  updateScrolled(); // run on load

  // ── Theme toggle ─────────────────────────────────────
  const html = document.documentElement;

  function getTheme(): string {
    return html.getAttribute('data-theme') || 'latte';
  }

  function applyThemeIcons(theme: string) {
    const isLatteD = document.getElementById('nav-theme-icon-latte');
    const isMochaD = document.getElementById('nav-theme-icon-mocha');
    const isLatteM = document.getElementById('nav-theme-icon-latte-m');
    const isMochaM = document.getElementById('nav-theme-icon-mocha-m');

    if (theme === 'mocha') {
      if (isLatteD) isLatteD.style.display = 'none';
      if (isMochaD) isMochaD.style.display = 'inline';
      if (isLatteM) isLatteM.style.display = 'none';
      if (isMochaM) isMochaM.style.display = 'inline';
    } else {
      if (isLatteD) isLatteD.style.display = 'inline';
      if (isMochaD) isMochaD.style.display = 'none';
      if (isLatteM) isLatteM.style.display = 'inline';
      if (isMochaM) isMochaM.style.display = 'none';
    }
  }

  function toggleTheme() {
    const next = getTheme() === 'mocha' ? 'latte' : 'mocha';
    document.body.classList.add('theme-transitioning');
    html.setAttribute('data-theme', next);
    localStorage.setItem('pk-theme', next);
    applyThemeIcons(next);
    setTimeout(() => document.body.classList.remove('theme-transitioning'), 300);
  }

  applyThemeIcons(getTheme());

  document.getElementById('nav-theme-btn')?.addEventListener('click', toggleTheme);
  document.getElementById('nav-theme-btn-mobile')?.addEventListener('click', toggleTheme);

  // ── Help popup ───────────────────────────────────────
  document.getElementById('nav-help-btn')?.addEventListener('click', () => {
    document.getElementById('kb-help')?.classList.toggle('hidden');
  });

  // ── Scroll-spy (IntersectionObserver) ────────────────
  const spySections = ['hero', 'experience', 'projects', 'contact'];
  const desktopLinks = document.querySelectorAll<HTMLAnchorElement>('.nav-link[data-spy]');
  const overlayLinks = document.querySelectorAll<HTMLAnchorElement>('.overlay-link[data-spy]');

  function setActive(sectionId: string) {
    [...desktopLinks, ...overlayLinks].forEach((link) => {
      link.dataset.active = link.dataset.spy === sectionId ? 'true' : 'false';
    });
  }

  const observer = new IntersectionObserver(
    (entries) => {
      entries.forEach((entry) => {
        if (entry.isIntersecting) {
          setActive(entry.target.id);
        }
      });
    },
    { threshold: 0.4 }
  );

  spySections.forEach((id) => {
    const el = document.getElementById(id);
    if (el) observer.observe(el);
  });

  // ── Smooth scroll on nav link click ──────────────────
  function scrollToId(id: string) {
    const el = document.getElementById(id);
    if (el) {
      el.scrollIntoView({ behavior: 'smooth', block: 'start' });
    }
  }

  [...desktopLinks, ...overlayLinks].forEach((link) => {
    link.addEventListener('click', (e) => {
      e.preventDefault();
      const target = link.getAttribute('href')?.replace('#', '') || '';
      scrollToId(target);
      // Close overlay if open
      closeOverlay();
    });
  });

  // ── Mobile overlay ────────────────────────────────────
  const overlay = document.getElementById('nav-overlay');
  const hamburger = document.getElementById('nav-hamburger');
  const hamOpen = document.getElementById('nav-hamburger-open');
  const hamClose = document.getElementById('nav-hamburger-close');

  function openOverlay() {
    overlay?.classList.add('open');
    overlay?.removeAttribute('aria-hidden');
    hamburger?.setAttribute('aria-expanded', 'true');
    if (hamOpen) hamOpen.style.display = 'none';
    if (hamClose) hamClose.style.display = 'inline';
    document.body.style.overflow = 'hidden';
  }

  function closeOverlay() {
    overlay?.classList.remove('open');
    overlay?.setAttribute('aria-hidden', 'true');
    hamburger?.setAttribute('aria-expanded', 'false');
    if (hamOpen) hamOpen.style.display = 'inline';
    if (hamClose) hamClose.style.display = 'none';
    document.body.style.overflow = '';
  }

  hamburger?.addEventListener('click', () => {
    overlay?.classList.contains('open') ? closeOverlay() : openOverlay();
  });
</script>
